## Criação do BD 
          CREATE SCHEMA login ;

          CREATE TABLE usuarios (
          id INT NOT NULL AUTO_INCREMENT,
          nome VARCHAR(45) NOT NULL ,
          email VARCHAR(45) NOT NULL,
          senha VARCHAR(45) NOT NULL,
          PRIMARY KEY (id));
